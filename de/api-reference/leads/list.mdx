---
title: "Leads auflisten"
api: "GET https://app.famulor.de/api/user/leads"
description: "Alle Leads für den authentifizierten Benutzer auflisten"
---


Dieser Endpunkt ermöglicht es dir, alle Leads des authentifizierten Benutzers mit Filter- und Paginierungsoptionen aufzulisten.

### Query Parameters

<ParamField query="status" type="string">
  Leads nach Status filtern. Mögliche Werte: `created`, `scheduled`, `processing`, `completed`, `rescheduled`, `reached-max-retries`, `blacklisted`
</ParamField>

<ParamField query="campaign_id" type="integer">
  Leads nach Kampagnen-ID filtern
</ParamField>

<ParamField query="phone_number" type="string">
  Leads nach Telefonnummer filtern (Teiltreffer unterstützt)
</ParamField>

<ParamField query="date_from" type="string">
  Leads filtern, die ab diesem Datum erstellt wurden (`YYYY-MM-DD` Format)
</ParamField>

<ParamField query="date_to" type="string">
  Leads filtern, die bis zu diesem Datum erstellt wurden (`YYYY-MM-DD` Format)
</ParamField>

<ParamField query="per_page" type="integer" default="15">
  Anzahl der Leads pro Seite (`1-100`)
</ParamField>

<ParamField query="page" type="integer" default="1">
  Seitennummer
</ParamField>

### Antwort-Felder

<ResponseField name="data" type="array">
  Liste der Leads

  <Expandable title="Lead-Eigenschaften">
    <ResponseField name="id" type="integer">
      Die ID des Leads
    </ResponseField>

    <ResponseField name="campaign_id" type="integer">
      Die ID der Kampagne, zu der dieser Lead gehört
    </ResponseField>

    <ResponseField name="phone_number" type="string">
      Die Telefonnummer des Leads im E.164-Format
    </ResponseField>

    <ResponseField name="variables" type="object">
      Die dem Lead zugeordneten Variablen
    </ResponseField>

    <ResponseField name="status" type="string">
      Der Status des Leads
    </ResponseField>

    <ResponseField name="created_at" type="string">
      Das Erstellungsdatum des Leads
    </ResponseField>

    <ResponseField name="updated_at" type="string">
      Das letzte Aktualisierungsdatum des Leads
    </ResponseField>

    <ResponseField name="campaign" type="object">
      Die Kampagnen-Informationen

      <Expandable title="Kampagnen-Eigenschaften">
        <ResponseField name="id" type="integer">
          Die ID der Kampagne
        </ResponseField>

        <ResponseField name="name" type="string">
          Der Name der Kampagne
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="secondary_contacts" type="array">
      Array der sekundären Kontakt-Leads, die mit diesem Lead verknüpft sind

      <Expandable title="Sekundäre Kontakt-Eigenschaften">
        <ResponseField name="id" type="integer">
          Die ID des sekundären Kontakt-Leads
        </ResponseField>

        <ResponseField name="phone_number" type="string">
          Die Telefonnummer des sekundären Kontakts im E.164-Format
        </ResponseField>

        <ResponseField name="variables" type="object">
          Die dem sekundären Kontakt zugeordneten Variablen
        </ResponseField>

        <ResponseField name="status" type="string">
          Der Status des sekundären Kontakts
        </ResponseField>

        <ResponseField name="created_at" type="string">
          Das Erstellungsdatum des sekundären Kontakts
        </ResponseField>

        <ResponseField name="updated_at" type="string">
          Das letzte Aktualisierungsdatum des sekundären Kontakts
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="current_page" type="integer">
  Die aktuelle Seitennummer
</ResponseField>

<ResponseField name="per_page" type="integer">
  Anzahl der Einträge pro Seite
</ResponseField>

<ResponseField name="total" type="integer">
  Gesamtanzahl der Leads, die den Kriterien entsprechen
</ResponseField>

<ResponseField name="last_page" type="integer">
  Die letzte Seitennummer
</ResponseField>
