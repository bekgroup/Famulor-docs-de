---
title: "List calls"
api: "GET /user/calls"
description: "Retrieve call history with filtering and pagination"
---

Get a list of all calls in your account with advanced filtering options to find specific calls.

### Query Parameters

<ParamField query="status" type="string" optional>
  Filter by call status. Options: `initiated`, `ringing`, `busy`, `in-progress`, `ended`, `completed`, `ended_by_customer`, `ended_by_assistant`, `no-answer`, `failed`
</ParamField>

<ParamField query="type" type="string" optional>
  Filter by call type. Options: `inbound`, `outbound`, `web`
</ParamField>

<ParamField query="phone_number" type="string" optional>
  Find calls to or from a specific phone number
</ParamField>

<ParamField query="assistant_id" type="integer" optional>
  Filter calls handled by a specific assistant
</ParamField>

<ParamField query="campaign_id" type="integer" optional>
  Filter calls related to a specific campaign
</ParamField>

<ParamField query="date_from" type="string" optional>
  Start date for date range filter (format: YYYY-MM-DD)
</ParamField>

<ParamField query="date_to" type="string" optional>
  End date for date range filter (format: YYYY-MM-DD)
</ParamField>

<ParamField query="per_page" type="integer" optional>
  Number of results per page (range: 1-100, default: 15)
</ParamField>

<ParamField query="page" type="integer" optional>
  Page number to retrieve (default: 1)
</ParamField>

### Response Fields

<ResponseField name="data" type="array">
  Array of call records
  <Expandable title="properties">
    <ResponseField name="id" type="integer">
      Unique call identifier
    </ResponseField>

    <ResponseField name="assistant_name" type="string">
      Name of the assistant that handled this call
    </ResponseField>

    <ResponseField name="campaign_name" type="string">
      Campaign name if this call was part of a campaign
    </ResponseField>

    <ResponseField name="type" type="string">
      Call direction type (`inbound`, `outbound`, or `web`)
    </ResponseField>

    <ResponseField name="duration" type="integer">
      Call length in seconds
    </ResponseField>

    <ResponseField name="assistant_phone_number" type="string">
      Phone number used by the assistant
    </ResponseField>

    <ResponseField name="client_phone_number" type="string">
      Customer's phone number
    </ResponseField>

    <ResponseField name="status" type="string">
      Final call status
    </ResponseField>

    <ResponseField name="transcript" type="string">
      Complete conversation transcript
    </ResponseField>

    <ResponseField name="variables" type="object">
      Data collected or passed during the call
    </ResponseField>

    <ResponseField name="evaluation" type="object">
      AI-generated call analysis and performance metrics
    </ResponseField>

    <ResponseField name="webhook_response" type="object">
      Response data from webhook callbacks (if configured)
    </ResponseField>

    <ResponseField name="carrier_cost" type="number">
      Telephony carrier charges for this call
    </ResponseField>

    <ResponseField name="total_cost" type="number">
      Total cost including all fees
    </ResponseField>

    <ResponseField name="answered_by" type="string">
      Type of answerer (`human`, `machine`, or `unknown`)
    </ResponseField>

    <ResponseField name="recording_url" type="string">
      URL to download call recording (if recording was enabled)
    </ResponseField>

    <ResponseField name="created_at" type="string">
      Call initiation timestamp
    </ResponseField>

    <ResponseField name="updated_at" type="string">
      Last update timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="current_page" type="integer">
  Current page number in pagination
</ResponseField>

<ResponseField name="per_page" type="integer">
  Number of items per page
</ResponseField>

<ResponseField name="total" type="integer">
  Total number of calls matching your filters
</ResponseField>

<ResponseField name="last_page" type="integer">
  Total number of pages available
</ResponseField>
