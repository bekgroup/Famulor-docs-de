a---
title: "Mid-Call-Tools auflisten"
api: "GET /user/tools"
description: "Alle Mid-Call-Tools abrufen"
---

Dieser Endpunkt ermöglicht es Ihnen, alle Mid-Call-Tools abzurufen. Mid-Call-Tools ermöglichen es Ihren KI-Assistenten, während eines Anrufs mit externen APIs zu interagieren.

### Header

<ParamField header="Authorization" type="string" required>
  Bearer-Token für die Authentifizierung
</ParamField>

<ParamField header="Content-Type" type="string" required>
  Muss `application/json` sein
</ParamField>

<ParamField header="Accept" type="string" required>
  Muss `application/json` sein
</ParamField>

### Antwortfelder

<ResponseField name="data" type="array">
  Array von Mid-Call-Tools
  <Expandable title="Eigenschaften">
    <ResponseField name="id" type="integer">
      Die eindeutige Kennung des Tools
    </ResponseField>

    <ResponseField name="name" type="string">
      Der Name des Tools (Kleinbuchstaben mit Unterstrichen)
    </ResponseField>

    <ResponseField name="description" type="string">
      Detaillierte Erklärung, wann und wie die KI dieses Tool verwenden soll
    </ResponseField>

    <ResponseField name="endpoint" type="string">
      Die API-Endpunkt-URL, die aufgerufen wird
    </ResponseField>

    <ResponseField name="method" type="string">
      HTTP-Methode (GET, POST, PUT, PATCH, DELETE)
    </ResponseField>

    <ResponseField name="timeout" type="integer">
      Request-Timeout in Sekunden (1-30)
    </ResponseField>

    <ResponseField name="headers" type="array">
      HTTP-Header, die mit der Anfrage gesendet werden
      <Expandable title="Header-Eigenschaften">
        <ResponseField name="name" type="string">
          Header-Name
        </ResponseField>
        <ResponseField name="value" type="string">
          Header-Wert
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="schema" type="array">
      Parameter, die die KI extrahieren und an den Endpunkt senden wird
      <Expandable title="Schema-Eigenschaften">
        <ResponseField name="name" type="string">
          Parameter-Name
        </ResponseField>
        <ResponseField name="type" type="string">
          Parameter-Typ (string, number, boolean)
        </ResponseField>
        <ResponseField name="description" type="string">
          Beschreibung, um der KI zu helfen, diesen Parameter zu extrahieren
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="created_at" type="string">
      ISO 8601 Zeitstempel, wann das Tool erstellt wurde
    </ResponseField>

    <ResponseField name="updated_at" type="string">
      ISO 8601 Zeitstempel, wann das Tool zuletzt aktualisiert wurde
    </ResponseField>
  </Expandable>
</ResponseField>
