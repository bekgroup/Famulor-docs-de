---
title: "Leads auflisten"
api: "GET https://app.famulor.de/api/user/leads"
icon: "list"
description: "Alle Leads für den authentifizierten Benutzer auflisten"
---

Dieser Endpunkt ermöglicht es Ihnen, alle Leads aufzulisten, die dem authentifizierten Benutzer gehören.

### Antwort-Felder

<ResponseField name="leads" type="array">
  Array aller Leads des Benutzers

  <Expandable title="Lead-Eigenschaften">
    <ResponseField name="id" type="integer">
      Die ID des Leads
    </ResponseField>

    <ResponseField name="campaign_id" type="integer">
      Die ID der Kampagne, zu der dieser Lead gehört
    </ResponseField>

    <ResponseField name="phone_number" type="string">
      Die Telefonnummer des Leads im E.164-Format
    </ResponseField>

    <ResponseField name="variables" type="object">
      Die dem Lead zugeordneten Variablen
    </ResponseField>

    <ResponseField name="status" type="string">
      Der Status des Leads
    </ResponseField>

    <ResponseField name="created_at" type="string">
      Das Erstellungsdatum des Leads
    </ResponseField>

    <ResponseField name="updated_at" type="string">
      Das letzte Aktualisierungsdatum des Leads
    </ResponseField>

    <ResponseField name="campaign" type="object">
      Die Kampagnen-Informationen

      <Expandable title="Kampagnen-Eigenschaften">
        <ResponseField name="id" type="integer">
          Die ID der Kampagne
        </ResponseField>

        <ResponseField name="name" type="string">
          Der Name der Kampagne
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="secondary_contacts" type="array">
      Array der sekundären Kontakt-Leads, die mit diesem Lead verknüpft sind

      <Expandable title="Sekundäre Kontakt-Eigenschaften">
        <ResponseField name="id" type="integer">
          Die ID des sekundären Kontakt-Leads
        </ResponseField>

        <ResponseField name="phone_number" type="string">
          Die Telefonnummer des sekundären Kontakts im E.164-Format
        </ResponseField>

        <ResponseField name="variables" type="object">
          Die dem sekundären Kontakt zugeordneten Variablen
        </ResponseField>

        <ResponseField name="status" type="string">
          Der Status des sekundären Kontakts
        </ResponseField>

        <ResponseField name="created_at" type="string">
          Das Erstellungsdatum des sekundären Kontakts
        </ResponseField>

        <ResponseField name="updated_at" type="string">
          Das letzte Aktualisierungsdatum des sekundären Kontakts
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
